[docs]    def cut_timesignal(self, number_of_first_spectra=1024,
                       werte_nr=400, save_avg=False):
        """
        Cuts the timesignal into pieces (before FFT).
        Cuts the timesignal in number_of_first_spectra splits and
        calculates the first spectrum out of it.

        .. important:: 
            This function effectively calculates the averaged spectrum.
            It calculates the number of timerelated spectra and averages them.
            
        .. caution:: 
            This is a generator and not a function!
            This means that it behaves like a iterator so you can loop over it.

        .. code-block:: python
            :linenos:
            :caption: Example

            for spectrum, frequency_span_array, k in cut_timesignal():
                if k == number_of_first_spectra:
                    plot(frequency_span_array, spectrum, name='Averaged')
                else:
                    plot(frequency_span_array, spectrum, name=k)

        :param number_of_first_spectra: Number of first spectra to recieve.
                [4096, 2048, 1024, 512, 256, 128, 64, 32, 16, 8, 4, 2]
        :type number_of_first_spectra: int, optional, default: 1024
        
        :param werte_nr:
        :type werte_nr: int, optional, default: 400
        
        :param save_avg:
        :type save_avg: bool, optional, default: True
        """
        filepath = self.filepath

        self.logger.debug("cut_timesignal(%s)", filepath)

        self.number_of_first_spectra = number_of_first_spectra

        steplength_timesignal = int(
            len(self.timesignal) / number_of_first_spectra)
        self.NFach = 1. * (len(self.timesignal) / self.RATE) / (
                number_of_first_spectra * self.chunksize / self.frequency)
        # relation between time available to time necessary for 1024 (400) point spektra

        # Splitting the timesignal into pieces to calculate PSD
        k = 0
        first_spectra_df = pd.DataFrame()
        frequency_span_array = np.array([])
        for k in np.arange(number_of_first_spectra):
            timesignal_part = self.timesignal[k * steplength_timesignal:
                                              (k + 1) * steplength_timesignal]
            first_spectrum, \
                frequency_span_array = self.calculate_first_spectrum(
                    timesignal_part, werte_nr)
            # Only count poisitive frequencies
            yield first_spectrum, frequency_span_array, k
            first_spectra_df['S%d' % k] = first_spectrum

        self.frequency_span_array = frequency_span_array
        first_spectra_df['Frequency'] = self.frequency_span_array

        if (save_avg
                and self.prefixes.get('short')
                and self.filepath):
            path, filename = self.extract_filename(self.filepath)
            self.save_to_dat(self.first_spectrum_avg,
                             self.frequency_span_array,
                             path, filename,
                             'Averaged_' + self.prefixes['short'])
            # self.mpl.plot_spectrum(frequency_span_array, first_spetrum_avg,
            #                       final = True,
            #                       color = self.mplcolor)
            # self.save_plot(path, filename, 'Averaged_'+self.prefixes['short'])

        self.finalize_first_spectrum(first_spectra_df)

        del timesignal_part
        del first_spectra_df

        self.logger.debug("Finished calculating averaged first spectrum")
        yield self.first_spectrum_avg, self.frequency_span_array, k + 1

[docs]    def calculate_first_spectrum(self, timesignal, werte_nr=400,
                                 plot_every=True):
        """
        Calculate First Spectrum and the corresponding Frequency Axis.

        .. important::
            * Filters the time signal
            * Downsamples filtered_timesignal (if downsample == True).
            * Subtracts mean value (offset).
            * Applies window function (hanning).
            * Calculates fft.
            * Normalizes First Spectrum.


        Normalization
        -------------

        .. math::
            :caption: Hanning Normalization Factor

            S_2^{hanning} = \sum_{}^{} {1}/{2}(1-cos(i 2 \pi / N))^2 = 3/8 N

        See additional info for normaization of first spectrum:
        
        * https://holometer.fnal.gov/GH_FFT.pdf
         - https://holometer.fnal.gov/GH_FFT.pdf#equation.24
         - https://holometer.fnal.gov/GH_FFT.pdf#equation.5
        * http://stackoverflow.com/questions/15147287/numpy-wrong-amplitude-of-fftd-array
        * http://www.mikrocontroller.net/attachment/16949/windows.pdf - some more information about window functions

        .. code-block:: python
            :caption: Compare FFT with measured spectrum

            first_spectrum_scipy = (abs(fft(final_array, axis=1))/np.sqrt(length_for_hanning))**2
            test = first_spectrum - first_spectrum_scipy
            print('Numpy-Scipy-Test: ',
                   np.allclose(0, test, atol=1e-10),
                   np.allclose(0, test, atol=1e-10))
            
        
        
        :param timesignal: The raw timesignal.
        :type timesignal: numpy.array
        
        :param werte_nr:
        :type werte_nr: int, optional, default: 400
        
        :param plot_every: deprecated
        :type plot_every: bool, optional, default: True
        """
        # Applying the Low Pass Filter
        filtered_timesignal, a, b = self.low_pass_filter(timesignal)

        # Calculate Fourier Transformation (fft)
        if self.downsample:
            # Reshape filtered_timesignal.
            gap = int(self.RATE / self.frequency)
            # gap is the number of points to leave out
            # to fit the real frequency
            help_array = np.reshape(filtered_timesignal,
                                    (int(len(filtered_timesignal) / gap),
                                     gap)).transpose()
            length_for_hanning = int(self.chunksize * self.NFach)
            if not self.shifting_method:
                help_array = np.array([help_array[0]])
                gap = 1
        else:
            help_array = np.array([filtered_timesignal])
            length_for_hanning = len(filtered_timesignal)
            gap = 1

        # eliminate offset
        offset = np.mean(help_array, axis=1)
        help_array = help_array - offset.reshape(gap, 1)

        # Create hanning window.
        hanning_window = np.hanning(length_for_hanning)
        # hanning_window is the window, normalized to one;
        # out: ndarray, shape(M,)
        # Apply hanning window.
        final_array = help_array * hanning_window
        del help_array

        # Calculate First Spectrum. NORMALIZATION!
        first_spectrum = abs(np.fft.fft(final_array, axis=1)) ** 2
        s2 = 3. / 8. * length_for_hanning
        # S_2(hanning) = sum((0.5*(1-cos(2pij/N)))^2) = 0.5^2 * 3/2 * N = 3/8 N
        if self.downsample:
            first_spectrum = first_spectrum * 2 / (self.frequency * s2)
        else:
            first_spectrum = first_spectrum * 2 / (self.RATE * s2)
        # https://holometer.fnal.gov/GH_FFT.pdf#equation.24

        # normalize! (https://holometer.fnal.gov/GH_FFT.pdf#equation.5)
        # S_2(hanning) = sum((0.5*(1-cos(2pij/N)))^2) = 0.5^2 * 3/2 * N = 3/8 N

        # http://stackoverflow.com/questions/15147287/numpy-wrong-amplitude-of-fftd-array
        # is somehow related and beeing used...
        # still not completely sure about the 4 and the frequency maybe from
        # (https://holometer.fnal.gov/GH_FFT.pdf), but than where did S_2 go
        # and why still 2 times more!?
        # some more information about window function:
        # http://www.mikrocontroller.net/attachment/16949/windows.pdf
        #        first_spectrum_scipy = (abs(fft(final_array, axis=1))/np.sqrt(length_for_hanning))**2
        #        test = first_spectrum - first_spectrum_scipy
        #        print 'Numpy-Scipy-Test\n\nnp.allclose(0, test, atol=1e-10):',
        #                                   np.allclose(0, test, atol=1e-10)
        # alternative would be scipy package
        # needs to be normalized to the legth of signal
        # http://stackoverflow.com/questions/15147287/numpy-wrong-amplitude-of-fftd-array
        # which has the same as hanning (therefore /length_for_hanning)
        first_spectrum = np.mean(first_spectrum, axis=0)
        del filtered_timesignal

        # Multiply First Spectrum with SR Factor -> [S_R/R]
        if self.factor:
            first_spectrum = first_spectrum * self.factor

        # Create Frequency Axis
        if self.downsample:
            timestep = 1. / self.frequency
        else:
            timestep = 1. / self.RATE
        frequency_span_array = np.fft.fftfreq(len(first_spectrum), timestep)

        # Delete first two points (due to FFT)
        if self.downsample:
            frequency_span_array_truncated = frequency_span_array[2 + 1:int(
                werte_nr * self.NFach) + 1]
            first_spectrum_truncated = first_spectrum[
                                       2 + 1:int(werte_nr * self.NFach) + 1]
        else:
            first_spectrum = pd.DataFrame(
                {'freq': frequency_span_array,
                 'S': first_spectrum})
            first_spectrum = first_spectrum.query('freq > 0')
            frequency_span_array_truncated = first_spectrum.freq.iloc[2:]
            first_spectrum_truncated = first_spectrum.S.iloc[2:]

        # shift one point away from 0 (+1) and delete first two points (2)
        del frequency_span_array
        del first_spectrum

        return first_spectrum_truncated, frequency_span_array_truncated

[docs]    def low_pass_filter(self, timesignal):
        """
        Applying the low pass filter to filter the timesignal.
 
        :param timesignal: The raw timesignal to be filtered.
        :type timesignal: numpy.array

        """
        # Design filter.
        nyq_freq = 0.5 * self.RATE

        if self.short_mode == 0:
            if self.filter_analog:
                ad = True
                norm_pass = float(
                    self.passband_factor) * self.frequency * 2 * np.pi
                # DOESN'T WORK --> one has to incorporated (somehow...)
            else:
                ad = False
                norm_pass = float(
                    self.passband_factor) * self.frequency / nyq_freq

            (b, a) = signal.butter(self.filter_order, norm_pass,
                                   btype=self.filter_type, analog=ad,
                                   output='ba')
            # Design an Nth order lowpass digital or analog Butterworth filter 
            # and return the filter coefficients in (b,a) 
            # For analog filters, Wn is an angular frequency (e.g. rad/s) 
            # [For digital filters, Wn is normalized from 0 to 1, where 1 is
            # the Nyquist frequency, pi radians/sample. (Wn is thus in 
            # half-cycles / sample.)]
        else:
            ad = bool(self.filter_analog)
            norm_pass = float(self.passband_factor) * self.frequency / nyq_freq
            # Passband edge frequency, normalized from 0 to 1
            # (1 corresponds to pi radians/sample)
            (b, a) = signal.butter(self.filter_order, norm_pass,
                                   self.filter_type, analog=ad, output='ba')
            # Design an Nth order lowpass digital or analog Butterworth filter 
            # and return the filter coefficients in (b,a) form
        # Output of filtered signal.
        return signal.lfilter(b, a, timesignal), a, b

        """Calculates the Histogram from the timesignal
        
        :param timesignal:
        :type timesignal: numpy.array

        :param bins:
        :type bins: int

        :param second: If True it uses only every second datapoint in the timesignal.

        :type second: bool, optional, default: True

        :param prefix: Save Prefix
        :type prefix: str, optional, default: ``histo_``
        """
        self.logger.debug("histogram(, %s, %s, %s)" % (bins, second, prefix))
        if second:
            self.logger.info('every second point in histogram...')
            timesignal = np.array(timesignal[::2])
            # take every second point for histogram
            # (otherwise memory error for big files)
        histo, bin_edges = np.histogram(timesignal,

[docs]    def extract_filename(self, filepath):
        """Extracting the path (and adding /DAQ/ to folder) and filename from the filepath.
        
        :param filepath: The absolute path to the file
        :type filepath: str
        """
        path = filepath.split(os.sep)  # Get List with dirs / filename
        filename = path.pop()  # Extract filename from path

        path.append('DAQ')  # Append DAQ to path

        fullpath = os.sep.join(path)  # Get absolute path to DAQ dir
        if not os.path.exists(fullpath):
            os.makedirs(fullpath)  # Create dir if not exists
        return fullpath, filename

        """Reading the data from h5py File.
        
        :param channel: the channel to read out.
        :type channel: str, optional, default: Vx
        """

        timesignal = np.array([])
        try:
            self.hdf5_f = h5py.File(self.filepath, 'r')
            timesignal = self.hdf5_f[channel]
            param = self.hdf5_f['parameter']
            rate = int(param.value[0, 1])
            avg = int(param.value[1, 1])
        except Exception as e:
            self.logger.error("Unable to read hdf5: %s", e)
            return []
        return timesignal, rate, avg
